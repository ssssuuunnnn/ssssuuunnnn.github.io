>  _用純前端技術製作一個專業的圖文選單工具，讓使用者能自訂文字、上傳背景並一鍵下載_

DEMO頁面：<https://kuokaikai.github.io/line_oa_menu_img_gen/index.html>

### 專案概述

這個專案是一個純前端的LINE OA圖文選單產生器，主要功能包括：

  * 🎨 多種模板選擇（2×2、1×3、3×1、2×3）
  * 🖼️ 支援背景圖片上傳與拖拽
  * 🎨 自訂顏色與文字描邊
  * 📱 響應式設計
  * 💾 一鍵下載PNG圖片



**技術棧：** HTML5 Canvas + CSS3 + Vanilla JavaScript

### Canvas繪圖引擎

### 核心繪製流程
    
    
    function drawMenu() {  
        const canvas = document.getElementById('menuCanvas');  
        const ctx = canvas.getContext('2d');  
        const template = templates[currentTemplate];  
          
        // 1. 清除畫布  
        ctx.fillStyle = '#ffffff';  
        ctx.fillRect(0, 0, canvas.width, canvas.height);  
          
        // 2. 繪製背景  
        if (backgroundType === 'image' && backgroundImage) {  
            drawBackgroundImage(ctx, canvas);  
        } else {  
            ctx.fillStyle = document.getElementById('bgColor').value;  
            ctx.fillRect(0, 0, canvas.width, canvas.height);  
        }  
          
        // 3. 繪製格線和文字  
        drawMenuGrid(ctx, template);  
    }

### 智慧文字處理
    
    
    function drawWrappedText(ctx, text, x, y, maxWidth) {  
        const words = text.split('');  
        let line = '';  
        const lines = [];  
          
        // 自動換行邏輯  
        for (let char of words) {  
            const testLine = line + char;  
            if (ctx.measureText(testLine).width > maxWidth && line !== '') {  
                lines.push(line);  
                line = char;  
            } else {  
                line = testLine;  
            }  
        }  
        lines.push(line);  
          
        // 繪製多行文字  
        const lineHeight = fontSize * 1.2;  
        lines.forEach((line, index) => {  
            ctx.fillText(line, x, y + index * lineHeight);  
        });  
    }

**拖拽上傳**
    
    
    function setupDragAndDrop(uploadArea) {  
        uploadArea.addEventListener('dragover', (e) => {  
            e.preventDefault();  
            uploadArea.classList.add('dragover');  
        });  
          
        uploadArea.addEventListener('drop', (e) => {  
            e.preventDefault();  
            uploadArea.classList.remove('dragover');  
              
            const files = e.dataTransfer.files;  
            if (files[0] && files[0].type.startsWith('image/')) {  
                loadImageFile(files[0]);  
            }  
        });  
    }

### 圖片處理
    
    
    function loadImageFile(file) {  
        const reader = new FileReader();  
        reader.onload = function(e) {  
            const img = new Image();  
            img.onload = function() {  
                backgroundImage = img;  
                drawMenu(); // 重新繪製  
            };  
            img.src = e.target.result;  
        };  
        reader.readAsDataURL(file);  
    }

### 關鍵特色功能

### 1\. 動態模板切換
    
    
    function setupTemplateButtons() {  
        const templateBtns = document.querySelectorAll('.template-btn');  
        templateBtns.forEach(btn => {  
            btn.addEventListener('click', () => {  
                currentTemplate = btn.dataset.template;  
                generateMenuInputs(); // 重新生成輸入框  
                drawMenu(); // 重新繪製  
            });  
        });  
    }

### 2\. 背景圖片適配

支援三種填充模式：填滿裁切、完整顯示、拉伸填滿
    
    
    function drawBackgroundImage(ctx, canvas) {  
        const fitMode = document.getElementById('bgFitMode').value;  
        const opacity = document.getElementById('bgOpacity').value;  
          
        ctx.save();  
        ctx.globalAlpha = opacity;  
          
        // 根據模式計算尺寸  
        const dimensions = calculateImageDimensions(backgroundImage, canvas, fitMode);  
        ctx.drawImage(backgroundImage, dimensions.x, dimensions.y, dimensions.width, dimensions.height);  
          
        ctx.restore();  
    }

### 3\. 一鍵下載功能
    
    
    function downloadImage() {  
        const canvas = document.getElementById('menuCanvas');  
        const link = document.createElement('a');  
          
        try {  
            link.download = `LINE_OA_圖文選單_${currentTemplate}_${Date.now()}.png`;  
            link.href = canvas.toDataURL('image/png', 1.0);  
              
            document.body.appendChild(link);  
            link.click();  
            document.body.removeChild(link);  
        } catch (error) {  
            alert('下載失敗，請重新嘗試');  
        }  
    }
